    print('Sample looks Gaussian (fail to reject H0)')
else:

result = anderson(dataset['medv'])

        print('%.3f: %.3f, data looks normal (fail to reject H0)' % (sl, cv))

        print('%.3f: %.3f, data does not look normal (reject H0)' % (sl, cv))
